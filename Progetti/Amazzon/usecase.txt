paeseConPiuAcquisti(i: Data, f: Data): Nazione [0..*]

    precondizioni: i <= f

    postcondizioni:

        N = {(n,a,q) | exists c,is acq_cit(a,c) and cit_naz(c,n) and istante(a,is) and i <= is <= f}

        T = {(t,n) | Nazione(n) and (t = sum_{(n1,a,q) in N} and n1 = n)}

        result = arg_max_{(t,n) in T}(t)
